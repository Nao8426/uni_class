\section*{問題2}
\noindent
更新式を
\begin{eqnarray*}
    \left(\begin{array}{c}w_1\\w_2\\w_0\end{array}\right)=\left(\begin{array}{c}w_1\\w_2\\w_0\end{array}\right)+y\left(\begin{array}{c}x_1\\x_2\\-1\end{array}\right)
\end{eqnarray*}
とおくと，更新過程は以下の13ステップとなった．
\begin{table}[H]
    \begin{center}
        \begin{tabular}{c|ccc}
            \hline
            更新回数 & $w_1$ & $w_2$ & $w_0$\\
            \hline \hline
            0 & 1.0 & 1.0 & 0.0\\
            1 & -1.0 & 2.0 & -1.0\\
            2 & 1.0 & 2.1 & 0.0\\
            3 & -1.0 & 2.0 & 1.0\\
            4 & 1.0 & 3.0 & 0.0\\
            5 & -1.0 & 2.9 & 1.0\\
            6 & 1.0 & 3.9 & 0.0\\
            7 & -1.0 & 3.8 & 1.0\\
            8 & 1.0 & 3.9 & 2.0\\
            9 & -1.0 & 3.8 & 3.0\\
            10 & 1.0 & 4.8 & 2.0\\
            11 & -1.0 & 4.7 & 3.0\\
            12 & 1.0 & 5.7 & 2.0\\
            13 & -1.0 & 5.6 & 3.0\\
            \hline
        \end{tabular}
    \end{center}
\end{table}
